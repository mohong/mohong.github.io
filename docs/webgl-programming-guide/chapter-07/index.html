#视点和视线

视点（eye point）：观察者所处的位置，视线的起点；

视线（viewing direction）：从视点出发沿着观察方向的射线；

观察目标点（look-at point）：即被观察目标所在的点，可以用来确定视线；

上方向（up direction）：最终绘制在屏幕上的影像中的向上的方向；

    const gl = getWebGLContext(canvas)

    // 初始化着色器
    if (!initShaders(gl, VSHADER_SOURCE, FSHADER_SOUCE)) {
      return
    }

    // 设置顶点位置

    // 指定绘图区域的背景颜色

    // Set the view matrix
    gl.uniformMatrix4fv(u_ViewMatrix, false, viewMatrix.elements);

    // Clear <canvas>

    // Draw the rectangle
    gl.drawArrays(gl.TRIANGLES, 0, n);

  }

  function initVertexBuffers(gl) {
    const verticesTexCoords = new Float32Array([

    gl.enableVertexAttribArray(a_Position);  // Enable the assignment of the buffer object

    const a_Color = gl.getAttribLocation(gl.program, 'a_Color');
    gl.vertexAttribPointer(a_Color, 3, gl.FLOAT, false, FSIZE * 6, FSIZE * 3);
    gl.enableVertexAttribArray(a_Color);  // Enable the assignment of the buffer object

  }

  let g_eyeX = 0.20, g_eyeY = 0.25, g_eyeZ = 0.25; // 视点

  function main() {
    const canvas = document.getElementById('webgl')

    document.onkeydown = function (ev) {
      keydown(ev, gl, n, u_ViewMatrix, viewMatrix);
    }
    draw(gl, n, u_ViewMatrix, viewMatrix);   // Draw
    
    function keydown (ev, gl, n, u_ViewMatrix, viewMatrix) {
      if(ev.keyCode == 39) { // The right arrow key was pressed
        g_eyeX += 0.01;
      } else if (ev.keyCode == 37) { // The left arrow key was pressed
        g_eyeX -= 0.01;
      } else { 
        return; 
      }
      draw(gl, n, u_ViewMatrix, viewMatrix);    
    }

    function draw(gl, n, u_ViewMatrix, viewMatrix) {
      // Set the matrix to be used for to set the camera view
      viewMatrix.setLookAt(g_eyeX, g_eyeY, g_eyeZ, 0, 0, 0, 0, 1, 0);
      
      // Pass the view projection matrix
      gl.uniformMatrix4fv(u_ViewMatrix, false, viewMatrix.elements);

      gl.clear(gl.COLOR_BUFFER_BIT);     // Clear <canvas>

      gl.drawArrays(gl.TRIANGLES, 0, n); // Draw the rectangle
    }
  }

  const VSHADER_SOURCE = 
    'attribute vec4 a_Position;\n' +

    draw(gl, n, u_ProjMatrix, projMatrix, nf);   // Draw

        case 37: g_near -= 0.01; break; // 按下左方向键
        case 38: g_far += 0.01; break;

#正确处理对象的前后关系

gl.enable(cap)

开启cap表示的功能（capability）。

参数：

名称说明
cap指定需要开启的功能，有可能是以下几个：
gl.DEPTH_TEST：隐藏面消除
gl.BLEND：混合
gl.POLYGON_OFFSET_FILL：多边形位移

返回值：无
